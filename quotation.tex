\RequirePackage
	{ csquotes }

\tl_set_eq:NN
	\caoutchouc_enquote
	\enquote

\DeclareDocumentCommand
	\enquote { m } {
		\caoutchouc_enquote {
			\textcolour
				{ quotation }
				{ #1 }}}

\DeclareDocumentCommand
	\mkcitation { m }
	{ #1\,\footnote }
\DeclareDocumentCommand
	\mktextquote { m m m m m m } {
		#1 \textcolour
			{ quotation }
			{ #2 #4 }
		#3 #6 #5 }
\DeclareDocumentCommand
	\mkblockquote { m m } {
		\color_group_begin:
			\colour
				{ quotation }
			#1 #2
		\color_group_end: }
\DeclareDocumentCommand
	\mkbegdispquote { m m }
	{ \color_group_begin:
		\colour
			{ quotation }}
\DeclareDocumentCommand
	\mkenddispquote { m m }
		{ #1 #2 \color_group_end: }
